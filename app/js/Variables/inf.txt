let и const- имеют блочную область видимаость, т.е. в пределах фигурных скобок.
var - имеет локальную область фидимость, т.е. область видимости в пределах функции.
var - переменная, может быть переопределена(не используется сейчас).
let - переменная, может быть переопределена.
const - неизменяемая контстанта, но может быть изменена через массивы и объекты:
const arr = [1, 2, 3];
arr.push(4);
console.log(arr);

const obj = {
  es: "es6",
};
obj.es = "es7";
console.log(obj);

// хостинг - всплытие переменных
let и const-не всплывают, т.е. не могут быть вызваны до их обявление, в отличии от var.
console.log(c);
var c = 5; - выведет undefined, т.к. интерпретатор проходит по коду находит переменную var и присваивает ей undefined,
после выполняет console.log(c), а после присваивает 5;
console.log(c); 
let c = 5; - выведет ошибку, т.к. интерпретатор проходит по коду, пытается выполнить console.log(c), но не находит с = let(или const),
т.к. переменная(константа) объявлена почле вызова функции.
